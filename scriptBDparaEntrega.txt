CREATE PROCEDURE [dbo].[sp_ConsultarBitacora]
    @desde DATETIME,
    @hasta DATETIME,
    @usuario VARCHAR(50) = NULL,
    @criticidad INT = 0
AS
BEGIN
    SELECT id_bitacora, usuario, fecha, evento, informacion, criticidad
    FROM BITACORA
    WHERE fecha >= @desde AND fecha <= @hasta
	-- Filtra solo si NO es 'all' y NO está vacío
      AND (@usuario = 'all' OR @usuario = '' OR usuario = @usuario)
      -- Si @criticidad es 0, trae todos los niveles
      AND (@criticidad = 0 OR criticidad = @criticidad)
    ORDER BY fecha DESC;
END
GO

CREATE PROCEDURE [dbo].[sp_InicializarIdioma]
    @id_idioma INT
AS
BEGIN
    -- Inserta una fila en TRADUCCION para cada ETIQUETA que exista,
    -- siempre y cuando no exista ya una traducción para ese idioma.
    INSERT INTO TRADUCCION (id_etiqueta, id_idioma, texto)
    SELECT E.id, @id_idioma, ''
    FROM ETIQUETA E
    WHERE NOT EXISTS (
        SELECT 1 FROM TRADUCCION T 
        WHERE T.id_etiqueta = E.id AND T.id_idioma = @id_idioma
    )
END
GO

CREATE PROCEDURE [dbo].[sp_ObtenerRaices]
AS
BEGIN

CREATE PROCEDURE [dbo].[sp_ObtenerTraduccion]
    @nombre_etiqueta NVARCHAR(50),
    @id_idioma INT
AS
BEGIN
    SELECT T.texto
    FROM TRADUCCION T
    INNER JOIN ETIQUETA E ON T.id_etiqueta = E.id
    WHERE E.nombre = @nombre_etiqueta 
      AND T.id_idioma = @id_idioma
END
GO

CREATE PROCEDURE [dbo].[sp_ObtenerTraduccionesPorIdioma]
    @id_idioma INT
AS
BEGIN
    SELECT E.id AS IdEtiqueta, E.nombre AS Etiqueta, ISNULL(T.texto, '') AS Texto
    FROM ETIQUETA E
    LEFT JOIN TRADUCCION T ON E.id = T.id_etiqueta AND T.id_idioma = @id_idioma
END
GO

CREATE PROCEDURE [dbo].[sp_ValidarUsuario]
    @nombre VARCHAR(50),
    @contrasena VARCHAR(255)
AS
BEGIN
    SELECT id, nombre, contrasena 
    FROM USUARIO 
    WHERE nombre = @nombre AND contrasena = @contrasena
END
