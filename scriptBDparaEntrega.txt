CREATE PROCEDURE [dbo].[sp_ConsultarBitacora]
    @desde DATETIME,
    @hasta DATETIME,
    @usuario VARCHAR(50) = NULL,
    @criticidad INT = 0
AS
BEGIN
    SELECT id_bitacora, usuario, fecha, evento, informacion, criticidad
    FROM BITACORA
    WHERE fecha >= @desde AND fecha <= @hasta
	-- Filtra solo si NO es 'all' y NO estß vacÝo
      AND (@usuario = 'all' OR @usuario = '' OR usuario = @usuario)
      -- Si @criticidad es 0, trae todos los niveles
      AND (@criticidad = 0 OR criticidad = @criticidad)
    ORDER BY fecha DESC;
END
GO

CREATE PROCEDURE [dbo].[sp_EliminarPermiso] @idPermiso INT 
AS 
BEGIN
DELETE 
FROM USUARIO_PERMISO 
WHERE id_permiso = @idPermiso; 
DELETE

END

CREATE PROCEDURE [dbo].[sp_ObtenerHijosPorPadre]
    @idPadre INT
AS
BEGIN
    -- Nos aseguramos de traer el ID del hijo como 'id_permiso'
    SELECT P.id_permiso, P.nombre, P.tipo
    FROM PERMISO P
    INNER JOIN PERMISO_HIJO PH ON P.id_permiso = PH.id_hijo
    WHERE PH.id_padre = @idPadre
END
GO
